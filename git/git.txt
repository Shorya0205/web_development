# Step 1: Navigate to correct folder (use quotes for spaces)
PS C:\Shorya Gupta\CODE WORKSPACE\Web_dev> cd "javascript/Projects/currency converter"

# Step 2: Initialize git in THIS folder
PS C:\...\currency converter> git init
Initialized empty Git repository in C:/Shorya Gupta/CODE WORKSPACE/Web_dev/javascript/Projects/currency converter/.git/

# Step 3: Add files
PS C:\...\currency converter> git add .

# Step 4: Commit
PS C:\...\currency converter> git commit -m "Initial commit: Currency converter project"

# Step 5: Add GitHub remote (create repo on GitHub first)
PS C:\...\currency converter> git remote add origin https://github.com/Shorya0205/currency-converter.git

# Step 6: Verify remote
PS C:\...\currency converter> git remote -v

# Step 7: Rename branch to main
PS C:\...\currency converter> git branch -M main

# Step 8: Push to GitHub
PS C:\...\currency converter> git push -u origin main
